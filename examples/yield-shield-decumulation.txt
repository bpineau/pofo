# Yield Shield — gentle 4%/yr decumulation demo
# Idea: a balanced, income-tilted portfolio drawn down at 4% a year. The
#   dividend + bond yield ('shield') covers much of the withdrawal, so fewer
#   shares are sold in downturns (less sequence-of-returns risk).
# Inspired by: Millennial Revolution 'Yield Shield'; Bengen/Trinity 4% rule.
#   https://www.millennial-revolution.com/invest/workshop-15-yield-shield/
# Demonstrates #meta capital + withdraw. UCITS.
#meta capital:500000
#meta withdraw:4%/year
#meta rebalance:180
#meta sim:on

30    VHYL    # High-dividend world (VHYL)
10    IWDP    # Global REITs (IWDP)
30    IEGA    # € Government Bond (IEGA)
10    ITPS    # TIPS (ITPS)
10    GOLD       # Gold (IGLN)
10    ERNA       # Cash / ultrashort (ERNA)
